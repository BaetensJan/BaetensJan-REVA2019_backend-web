import {AfterViewInit, ChangeDetectorRef, Component} from '@angular/core';
import {Categorie} from "../models/categorie.model";
import {Router} from "@angular/router";
import {CategorieShareService} from "./categorie-share.service";
import {CategorieenDataService} from "../categorieen/categorieen-data.service";

@Component({
  selector: 'app-categorie',
  templateUrl: './categorie.component.html',
  styleUrls: ['./categorie.component.css']
})
export class CategorieComponent implements AfterViewInit {
  /**
   * @ignore
   */
  public selectedCategorie: Categorie;
  /**
   * @ignore
   */
  public heeftWaarde = false;
  /**
   * @ignore
   */
  public aanpassen: boolean;

  /**
   * Constructor
   *
   * @param router
   * @param _categorieShareService
   * @param _categorieenDataSevice
   * @param cdRef
   */
  constructor(
    private router: Router,
    private _categorieShareService: CategorieShareService,
    private _categorieenDataSevice: CategorieenDataService,
    private cdRef: ChangeDetectorRef
  ) {
  }

  /**
   * Loads category or creates empty category based on editing or creating
   */
  ngAfterViewInit() {
    this.selectedCategorie = this._categorieShareService.categorie;
    this.aanpassen = this._categorieShareService.aanpassen;
    if (this.selectedCategorie == undefined) {
      this.selectedCategorie = new Categorie();
      this.selectedCategorie.name = "";
      this.selectedCategorie.description = "";
    }
    this.heeftWaarde = true;
    this.cdRef.detectChanges();
  }

  /**
   * Click event to cancel
   */
  onAnnuleer() {
    this.router.navigate(["/categorieen"]);
  }

  /**
   * Click event to save or create Category
   */
  onOpslaan() {
    if (this.selectedCategorie.id == undefined) {
      this._categorieenDataSevice.voegCategorieToe(this.selectedCategorie).subscribe(categorie => {
        this.router.navigate(["/categorieen"]);
      });
    } else {
      this._categorieenDataSevice.updateCategorie(this.selectedCategorie).subscribe(categorie => {
        this.router.navigate(["/categorieen"]);
      });
    }
  }

}

<div *ngIf="heeftWaarde" class="form-container">
  <mat-form-field>
    <input matInput [(ngModel)]="selectedCategorie.name" placeholder="Categorie naam">
  </mat-form-field>

  <mat-form-field>
    <textarea matInput [(ngModel)]="selectedCategorie.description" placeholder="Beschrijving"></textarea>
  </mat-form-field>

  <div>
    <button class="btn btn-danger pull-left" (click)="onAnnuleer()">Annuleer</button>
    <button class="btn btn-success pull-right" (click)="onOpslaan()">Opslaan</button>
  </div>
</div>
