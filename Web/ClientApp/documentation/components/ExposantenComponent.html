import {Component, ElementRef, ViewChild} from '@angular/core';
import {MatPaginator, MatSort} from "@angular/material";
import {BsModalRef, BsModalService} from "ngx-bootstrap";
import {Router} from "@angular/router";
import {ExposantenDataService} from "./exposanten-data.service";
import {ExposantShareService} from "../exposant/exposant-share.service";
import {Exposant} from "../models/exposant.model";
import {ExposantenDatasource} from "./exposanten-datasource";

@Component({
  selector: 'app-exposanten',
  templateUrl: './exposanten.component.html',
  styleUrls: ['./exposanten.component.css']
})
export class ExposantenComponent {
  /**
   * @ignore
   */
  @ViewChild(MatPaginator) paginator: MatPaginator;
  /**
   * @ignore
   */
  @ViewChild(MatSort) sort: MatSort;
  /**
   * @ignore
   */
  @ViewChild('modal') modal: ElementRef;
  /**
   * @ignore
   */
  dataSource: ExposantenDatasource;
  /**
   * @ignore
   */
  data: Exposant[] = [];
  /**
   * @ignore
   */
  clickedItem: Exposant;
  /**
   * @ignore
   */
  refModal: BsModalRef;

  /**
   * Columns displayed in the table. Columns IDs can be added, removed, or reordered.
   *
   */
  displayedColumns = ['name', 'buttons'];

  /**
   * Constructor
   *
   * @param router: Router
   * @param modalService: BsModalService
   * @param _exposantDataService: ExposantenDataService
   * @param _exposantShareService: ExposantenShareService
   */
  constructor(private router: Router,
              private modalService: BsModalService,
              private _exposantenDataService: ExposantenDataService,
              private _exposantShareService: ExposantShareService
  ) {
    this.exposanten();
  }

  /**
   * Haalt exposanten op van de backend en geeft door aan Datasource om table te seeden.
   *
   */
  exposanten() {
    this._exposantenDataService.exposanten.subscribe(exposanten => {
      this.data = exposanten as Exposant[];
      this.dataSource = new ExposantenDatasource(this.data, this.paginator, this.sort);
    });
  }

  /**
   * Click event om exposant te verwijderen. Geeft de exposant door aan de modal voor bevestiging.
   *
   * @param row: Exposant
   */
  onVerwijderExposant(row: Exposant) {
    this.clickedItem = row;
    this.refModal = this.modalService.show(this.modal);
  }

  /**
   * Toggle voor modal te tonen.
   *
   */
  hideModal() {
    this.refModal.hide();
  }

  /**
   * Click event om exposant toe te voegen. Geeft een lege exposant door aan share service om zo de data te kunnen verkrijgen in het exposant component.
   *
   */
  onToevoegenExposant() {
    this._exposantShareService.exposant = null;
    this.router.navigate((["/exposant"]))
  }

  /**
   * Click event om exposant aan te passen. Geeft de exposant door aan share service om zo de data te kunnen verkrijgen in het exposant component.
   *
   */
  onAanpassenExposant(row: Exposant) {
    this._exposantShareService.exposant = row;
    this._exposantShareService.aanpassen = true;
    this.router.navigate(["/exposant"]);
  }

  /**
   * Bevestiging van het verwijderen van een exposant. Dit wordt gebruikt bij het modal om te bevestigen.
   *
   */
  verwijderenBevestigd() {
    console.log("verwijderen van " + this.clickedItem.name);
    this.refModal.hide();
    this._exposantenDataService.verwijderExposant(this.clickedItem).subscribe(exposant => {
      let index = this.data.indexOf(this.data.find((c) => c.id === exposant.id));
      this.data.splice(index, 1);
      this.dataSource = new ExposantenDatasource(this.data, this.paginator, this.sort);
    });
  }


}
