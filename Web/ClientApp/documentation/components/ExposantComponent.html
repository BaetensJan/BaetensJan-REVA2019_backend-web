import {AfterViewInit, ChangeDetectorRef, Component, ElementRef, OnInit, ViewChild} from '@angular/core';
import {Router} from "@angular/router";
import {Exposant} from "../models/exposant.model";
import {ExposantShareService} from "./exposant-share.service";
import {ExposantenDataService} from "../exposanten/exposanten-data.service";
import {CategorieenDataService} from "../categorieen/categorieen-data.service";
import {Categorie} from "../models/categorie.model";

@Component({
  selector: 'app-exposant',
  templateUrl: './exposant.component.html',
  styleUrls: ['./exposant.component.css']
})
export class ExposantComponent implements OnInit, AfterViewInit {
  /**
   * @ignore
   */
  @ViewChild('myCanvas', {read: ElementRef}) public canvas: ElementRef;
  /**
   * @ignore
   */
  @ViewChild('myContainer', {read: ElementRef}) public container: ElementRef;
  /**
   * @ignore
   */
  public cx: CanvasRenderingContext2D;
  /**
   * @ignore
   */
  public selectedExposant: Exposant;
  /**
   * @ignore
   */
  private _categorie: Categorie;
  /**
   * @ignore
   */
  public heeftWaarde = false;
  /**
   * @ignore
   */
  public aanpassen: boolean;
  /**
   * @ignore
   */
  public categorieen: Categorie[] = [];

  /**
   * Constructor
   *
   * @param router
   * @param _exposantShareService
   * @param _exposantenDataService
   * @param _categorieenDataService
   * @param cdRef
   */
  constructor(
    private router: Router,
    private _exposantShareService: ExposantShareService,
    private _exposantenDataService: ExposantenDataService,
    private _categorieenDataService: CategorieenDataService,
    private cdRef: ChangeDetectorRef
  ) {
  }

  /**
   * Gets all categories and resets the loaded exhibitor
   */
  ngOnInit() {
    this.selectedExposant = new Exposant();
    this.selectedExposant.name = "";
    this._categorieenDataService.categorieen.subscribe(categorieen => {
      this.categorieen = categorieen;
    });
  }

  /**
   * loads Canvas on view init and shows the position of the exhibitor
   */
  ngAfterViewInit() {
    this.cx = (<HTMLCanvasElement>this.canvas.nativeElement).getContext('2d');
    let img = new Image;
    //after image has been loaded execute this:
    img.onload = () => {
      //x = scale element for the canvas
      let x = (<HTMLElement>this.container.nativeElement).clientWidth / img.width;
      this.canvas.nativeElement.width = img.width * x;
      this.canvas.nativeElement.height = img.height * x;
      //draw image on canvas
      this.cx.drawImage(img, 0, 0, img.width * x, img.height * x);
      this._categorieenDataService.categorieen.subscribe(categorieen => {
        this.categorieen = categorieen;
        this.selectedExposant = this._exposantShareService.exposant;
        this.aanpassen = this._exposantShareService.aanpassen;
        if (this.selectedExposant == undefined) {
          this.selectedExposant = new Exposant();
          this.selectedExposant.name = "";
        } else {
          this.categorieen.forEach(categorie => {
            if (categorie.id === this.selectedExposant.category.id) {
              this._categorie = categorie;
              this.cdRef.detectChanges();
            }
          });
          //get position of element and execute draw function
          let pos = {
            x: this.selectedExposant.x * this.canvas.nativeElement.width,
            y: this.selectedExposant.y * this.canvas.nativeElement.height
          };
          this.draw(pos);
        }
        this.heeftWaarde = true;
        this.cdRef.detectChanges();
      });
    };
    img.src = "../../assets/beursmap.jpg";
  }

  /**
   * Click event to change position of exhibitor on canvas
   * @param e
   */
  onDraw(e) {
    let canvas = <HTMLCanvasElement>this.canvas.nativeElement;
    this.draw(this.getMousePos(canvas, e));
  }

  /**
   * Reset image and draw on location of param pos
   *
   * @param pos
   */
  draw(pos) {
    this.cx = (<HTMLCanvasElement>this.canvas.nativeElement).getContext('2d');
    let img = new Image;
    img.onload = () => {
      let x = (<HTMLElement>this.container.nativeElement).clientWidth / img.width;
      this.canvas.nativeElement.width = img.width * x;
      this.canvas.nativeElement.height = img.height * x;
      this.cx.drawImage(img, 0, 0, img.width * x, img.height * x);
      let posx = pos.x;
      let posy = pos.y;
      this.selectedExposant.x = (pos.x / this.canvas.nativeElement.width);
      this.selectedExposant.y = (pos.y / this.canvas.nativeElement.height);
      this.cx.fillStyle = "#000000";
      this.cx.beginPath();
      this.cx.arc(posx, posy, 15, 0, 2 * Math.PI);
      this.cx.fill();
    };
    img.src = "../../assets/beursmap.jpg";
  }

  /**
   * get mouse position compared to canvas and client screen
   *
   * @param canvas
   * @param evt
   */
  getMousePos(canvas, evt) {
    let rect = canvas.getBoundingClientRect();
    return {
      x: evt.clientX - rect.left,
      y: evt.clientY - rect.top
    };
  }

  /**
   * Click event to cancel
   */
  onAnnuleer() {
    this.router.navigate(["/exposanten"]);
  }

  /**
   * Click event to save/create an exhibitor
   */
  onOpslaan() {
    if (this.selectedExposant.id == undefined) {
      this._exposantenDataService.VoegExposantToe(this.selectedExposant).subscribe(exposant => {
        console.log(exposant);
        this.router.navigate(["/exposanten"]);
      });
    } else {
      this._exposantenDataService.updateExposant(this.selectedExposant).subscribe(exposant => {
        this.router.navigate(["/exposanten"]);
      });
    }
  }

  /**
   * Getter for Category
   */
  get categorie() {
    return this._categorie;
  }

  /**
   * Setter for Category
   *
   * @param value
   */
  set categorie(value: Categorie) {
    this.selectedExposant.category = value;
    this._categorie = value;
  }
}
