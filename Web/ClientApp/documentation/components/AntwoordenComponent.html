import {Component, ElementRef, ViewChild} from '@angular/core';
import * as jsPDF from 'jspdf';
import {Group} from "../models/group.model";
import {MatPaginator, MatSort} from "@angular/material";
import {AntwoordenDataSource} from "./antwoorden-datasource";
import {BsModalService} from "ngx-bootstrap";

import {AntwoordenDataService} from "./antwoorden-data.service";

@Component({
  selector: 'app-antwoorden',
  templateUrl: './antwoorden.component.html',
  styleUrls: ['./antwoorden.component.css']
})
export class AntwoordenComponent {
  /**
   * @ignore
   */
  @ViewChild(MatPaginator) paginator: MatPaginator;
  /**
   * @ignore
   */
  @ViewChild(MatSort) sort: MatSort;
  /**
   * @ignore
   */
  @ViewChild('modal') modal: ElementRef;
  /**
   * @ignore
   */
  dataSource: AntwoordenDataSource;
  /**
   * @ignore
   */
  data: Group[];

  /** Columns displayed in the table. Columns IDs can be added, removed, or reordered. */
  displayedColumns = ['Groepsnaam', 'Download'];

  /**
   * Constructor
   *
   * @param _AntwoordenDataService
   * @param modalService
   */
  constructor(
    private _AntwoordenDataService: AntwoordenDataService,
    private modalService: BsModalService
  ) {
    this.groups();
  }

  /**
   * Gets all finished assignments from the backend for each group and adds this to the table
   */
  groups() {
    this._AntwoordenDataService.antwoorden.subscribe(antwoorden => {
      this.data = antwoorden as Group[];
      this.dataSource = new AntwoordenDataSource(this.data, this.paginator, this.sort);
    });
  }

  /**
   * Generates pdf file for group and downloads pdf
   *
   * @param row
   */
  pdfDownload(row: Group) {
    const doc = new jsPDF();
    doc.text("Groep: " + row.name, 10, 10);
    doc.text("aantal bezochte standen: " + row.finishedAssignments.length, 10, 20);
    let i = 0;
    // item.finishedAssignments.forEach(f => {
    row.finishedAssignments.forEach(f => {
      if (i == 0) {
        if (i > 0) {
          doc.addPage();
        }
        doc.text("stand " + (i + 1) + " :" + f.exhibitor.name, 10, 40);
        doc.text("categorie: " + f.exhibitor.category.name, 10, 50)
        doc.text(f.questions, 10, 60);
        doc.text(f.answers, 10, 70);
        doc.text("notities: " + f.notes, 10, 80);
        doc.addImage("../../assets/" + f.photo, "JPEG", 10, 90, 100, 100);
        i++;
      } else {
        if (i > 0) {
          doc.addPage();
        }
        doc.text("stand " + (i + 1) + " :" + f.exhibitor.name, 10, 10);
        doc.text("categorie: " + f.exhibitor.category.name, 10, 20)
        doc.text(f.questions, 10, 30);
        doc.text(f.answers, 10, 40);
        doc.text("notities: " + f.notes, 10, 50);
        doc.addImage("../../assets/" + f.photo, "JPEG", 10, 60, 100, 100);
        i++;
      }
    });
    doc.save(row.name + "_antwoorden.pdf");
  }
