import {Component, ViewChild} from '@angular/core';
import {MatPaginator, MatSort, MatTableDataSource} from '@angular/material';
import {GroepenDataSource} from './groepen-datasource';
import {animate, state, style, transition, trigger} from '@angular/animations';
import {GroepenDataService} from "./groepen-data.service";
import {Group} from "../models/group.model";
import {Observable} from "rxjs/Rx";

@Component({
  selector: 'groepen',
  templateUrl: './groepen.component.html',
  styleUrls: ['./groepen.component.css'],
  animations: [
    trigger('detailExpand', [
      state('collapsed', style({height: '0px', minHeight: '0', display: 'none'})),
      state('expanded', style({height: '*'})),
      transition('expanded <=> collapsed', animate('225ms cubic-bezier(0.4, 0.0, 0.2, 1)')),
    ]),
  ],
})
export class GroepenComponent {
  /**
   * @ignore
   */
  @ViewChild(MatPaginator) paginator: MatPaginator;
  /**
   * @ignore
   */
  @ViewChild(MatSort) sort: MatSort;
  /**
   * @ignore
   */
  public data: GroepenDataSource;
  /**
   * @ignore
   */
  public dataSource: MatTableDataSource<Group>;

  /** Columns displayed in the table. Columns IDs can be added, removed, or reordered. */
  displayedColumns = ['id', 'name', 'active'];
  /**
   * @ignore
   */
  expandedElement = 'leerlingen';

  /**
   * Applying filter for table
   *
   * @param filterValue
   */
  applyFilter(filterValue: string) {
    this.dataSource.filter = filterValue.trim().toLowerCase();
  }

  /**
   * Constructor
   *
   * @param _groepenDataService
   */
  constructor(private _groepenDataService: GroepenDataService) {
    this._groepenDataService.groups.subscribe(groepen => {
      this.data = new GroepenDataSource(this.paginator, this.sort, groepen);
      this.dataSource = new MatTableDataSource<Group>(groepen);
    });
  }

  /**
   * Getter for groups
   */
  get groups(): Observable<Group[]> {
    return this._groepenDataService.groups;
  }
}
