import {Exposant} from "./exposant.model";

export class Assignment {
  /**
   * @ignore
   */
  private _questions: string;
  /**
   * @ignore
   */
  private _answers: string;
  /**
   * @ignore
   */
  private _notes: string;
  /**
   * @ignore
   */
  private _photo: string;
  /**
   * @ignore
   */
  private _exhibitor: Exposant;

  /**
   * Constructor
   *
   * @param questions
   * @param answers
   * @param notes
   * @param photo
   * @param exhibitor
   */
  constructor(
    questions: string,
    answers: string,
    notes: string,
    photo: string,
    exhibitor: Exposant,
  ) {
    this._questions = questions;
    this._answers = answers;
    this._notes = notes;
    this._exhibitor = exhibitor;
    this._photo = photo;
  }

  /**
   * Static JSON to Object parser
   *
   * @param json
   */
  static fromJSON(json: any): Assignment {
    const rec = new Assignment(
      json.questions,
      json.answers,
      json.notes,
      json.exhibitor,
      json.photo
    );
    return rec;
  }

  /**
   * Object to JSON parser
   */
  toJSON() {
    return {
      questions: this._questions,
      answers: this._answers,
      notes: this._notes,
      exhibitor: this._exhibitor,
      photo: this._photo,
    };
  }

  /**
   * Getter Questions
   */
  get questions(): string {
    return this._questions;
  }

  /**
   * Getter for Answers
   */
  get answers(): string {
    return this._answers;
  }

  /**
   * Getter for Notes
   */
  get notes(): string {
    return this._notes;
  }

  /**
   * Getter for Exhibitor
   */
  get exhibitor(): Exposant {
    return this._exhibitor;
  }

  /**
   * Getter for Photo
   */
  get photo(): string {
    return this._photo;
  }
}
