import {Assignment} from "./assignment.model";

export class Group {
  /**
   * @ignore
   */
  private _id: string;
  /**
   * @ignore
   */
  private _name: string;
  /**
   * @ignore
   */
  private _finishedAssignments: Assignment[];
  /**
   * @ignore
   */
  private _members: String[];

  /**
   * Constructor
   *
   * @param name
   * @param finishedAssignments
   */
  constructor(
    name: string,
    finishedAssignments: Assignment[],
    //members: String[]
  ) {
    this._name = name;
    this._finishedAssignments = finishedAssignments;
    //this._members = members;
  }

  /**
   * Static JSON to Object parser
   *
   * @param json
   */
  static fromJSON(json: any): Group {
    const rec = new Group(
      json.name,
      json.finishedAssignments,
      // json.members
    );

    rec._id = json.id;
    return rec;
  }

  /**
   * Object to JSON converter
   */
  toJSON() {
    return {
      id: this._id,
      name: this._name,
      finishedAssignments: this._finishedAssignments,
      // members: this._members
    };
  }

  /**
   * Getter for Id
   */
  get id(): string {
    return this._id;
  }

  /**
   * Getter for name
   */
  get name(): string {
    return this._name;
  }

  /**
   * Getter for FinishedAssignments
   */
  get finishedAssignments(): Assignment[] {
    return this._finishedAssignments;
  }

  /*get members(): String[]{
    return this._members;
  }
  */

  /**
   * Setter for FinishedAssignments
   */
  set finishedAssignments(value: Assignment[]) {
    this._finishedAssignments = value;
  }
}
