  /**
   * Getter for Groups
   */
  get groups(): Observable<Group[]> {
    return this.http
      .get(`${this._appUrl}/Groups/`)
      .pipe(map((list: any[]): Group[] => list.map(Group.fromJSON)));
  }

  /**
   * Makes call to the backend and adds new Group
   *
   * @param group
   */
  addNewGroup(group: Group): Observable<Group> {
    return this.http
      .post(`${this._appUrl}/addGroup/`, group)
      .pipe(map(Group.fromJSON));
  }

  /**
   * Makes call to the backend and removes a group
   * @param rec
   */
  removeRecipe(rec: Group): Observable<Group> {
    return this.http
      .delete(`${this._appUrl}/removeGroup/${rec.id}`)
      .pipe(map(Group.fromJSON));
  }

  /**
   * Makes call to the backend and returns a group
   *
   * @param id
   */
  getRecipe(id: string): Observable<Group> {
    return this.http
      .get(`${this._appUrl}/getGroup/${id}`)
      .pipe(map(Group.fromJSON));
  }

}
