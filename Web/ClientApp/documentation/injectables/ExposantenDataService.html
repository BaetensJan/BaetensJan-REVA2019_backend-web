import {Injectable} from '@angular/core';
import {HttpClient} from "@angular/common/http";
import {Observable} from "rxjs";
import {map} from "rxjs/operators";
import {Exposant} from "../models/exposant.model";

@Injectable({
  providedIn: 'root'
})
export class ExposantenDataService {
  /**
   * Base url of the to connected connection
   */
  private readonly _appUrl = '/API/Exhibitor';

  /**
   * Constructor
   *
   * @param http
   */
  constructor(private http: HttpClient) {
  }

  /**
   * Makes call to backend and returns all exhibitors
   */
  get exposanten(): Observable<Exposant[]> {
    return this.http
      .get(`${this._appUrl}/Exhibitors/`)
      .pipe(map((list: any[]): Exposant[] => list.map(Exposant.fromJSON)));
  }

  /**
   * Makes call to backend and adds an exhibitor
   *
   * @param exposant
   * @constructor
   */
  VoegExposantToe(exposant: Exposant): Observable<Exposant> {
    console.log(exposant);
    return this.http
      .post(`${this._appUrl}/AddExhibitor/`, exposant)
      .pipe(map(Exposant.fromJSON));
  }

  /**
   * Makes call to the backend and removes an exhibitor
   *
   * @param rec
   */
  verwijderExposant(rec: Exposant): Observable<Exposant> {
    return this.http
      .delete(`${this._appUrl}/RemoveExhibitor/${rec.id}`)
      .pipe(map(Exposant.fromJSON));
  }

  /**
   * Makes call to the backend and updates an existing exhibitor
   *
   * @param rec
   */
  updateExposant(rec: Exposant): Observable<Exposant> {
    return this.http
      .post(`${this._appUrl}/UpdateExhibitor/`, rec)
      .pipe(map(Exposant.fromJSON));
  }
}
